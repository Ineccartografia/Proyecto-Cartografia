import networkx as nx
import folium
import osmnx as ox

provincias = [
    "Guayas, Ecuador",
    "Los Ríos, Ecuador",
    "Santa Elena, Ecuador",
    "Manabí, Ecuador",
    "Santo Domingo de los Tsáchilas, Ecuador",
    "Bolívar, Ecuador",
    "El Oro, Ecuador"
    ]
main_roads_types = ['motorway', 'trunk', 'primary', 'secondary', 'motorway_link', 'trunk_link', 'primary_link']

custom_filter_str = '["highway"~"' + '|'.join(main_roads_types) + '"]'

grafo = ox.graph_from_place(provincias, custom_filter=custom_filter_str)
ox.save_graphml(grafo, "zonal.graphml")
G = ox.load_graphml("zonal.graphml")
